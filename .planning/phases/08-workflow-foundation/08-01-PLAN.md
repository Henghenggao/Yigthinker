from __future__ import annotations
import json
import os
import tempfile
from datetime import datetime, timezone

from filelock import FileLock

**Class: WorkflowRegistry**
- `__init__(self, base_dir: Path | None = None)` -- defaults to `Path.home() / ".yigthinker" / "workflows"`. Store `self._base_dir`, `self._index_path = base_dir / "registry.json"`, `self._lock = FileLock(str(self._index_path) + ".lock", timeout=10)`. Per D-07, registry is global (process-scoped).
- `_ensure_dirs(self)` -- `self._base_dir.mkdir(parents=True, exist_ok=True)`
- `load_index(self) -> dict` -- If index_path missing, return `{"workflows": {}, "suppressed_suggestions": []}`. Otherwise `json.loads(self._index_path.read_text(encoding="utf-8"))`.
- `save_index(self, data: dict) -> None` -- Per D-06: acquire `self._lock`, write to `tempfile.mkstemp(dir=str(self._base_dir), suffix=".tmp")`, then `os.replace(tmp, str(self._index_path))`. Wrap in try/except to unlink tmp on failure.
- `next_version(self, name: str) -> int` -- Load manifest for name; if None return 1; else return `len(manifest["versions"]) + 1`. Per D-05, sequential integers.
- `get_manifest(self, name: str) -> dict | None` -- Read `self._base_dir / name / "manifest.json"`. Return None if missing.
- `save_manifest(self, name: str, manifest: dict) -> None` -- Atomic write to `self._base_dir / name / "manifest.json"` using same lock+tmp+replace pattern.
- `create(self, name: str, description: str, version_data: dict) -> Path` -- Per D-08:
  1. version = self.next_version(name)
  2. version_dir = self._base_dir / name / f"v{version}"
  3. version_dir.mkdir(parents=True, exist_ok=True)
  4. Build version entry: `{"version": version, "created_at": datetime.now(timezone.utc).isoformat(), "description": description, "files": list(version_data.keys())}`
  5. Write each file in version_data (key=filename, value=content) to version_dir
  6. Load or create manifest, append version entry to `manifest["versions"]`, save manifest
  7. Update registry.json index entry for this workflow: `{"status": "active", "latest_version": version, "description": description, "created_at": ..., "updated_at": ...}`
  8. Return version_dir
- `list_workflows(self) -> list[dict]` -- Load index, return list of workflow entries with name added.
- `update(self, name: str, description: str, version_data: dict, changelog: str = "") -> Path` -- Per D-11: Same as create but also writes changelog entry to `self._base_dir / name / "changelog.txt"` (append mode). Previous version directory untouched.
