```python
"""Power Automate guided-mode bundle builder.

Packs the 3 PA templates into flow_import.zip with PA's expected paths.
Per Pattern 5 research: definition.json must live at
    Microsoft.Flow/flows/<workflow_name>/definition.json
"""
from __future__ import annotations

import zipfile
from pathlib import Path
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from yigthinker.tools.workflow.template_engine import TemplateEngine


def build_pa_bundle(
    *,
    workflow_name: str,
    variables: dict,
    engine: "TemplateEngine",
    output_dir: Path,
) -> Path:
    """Assemble flow_import.zip and return its path.

    variables must include:
        display_name, description, cron_expression,
        recurrence_frequency, recurrence_interval, registration_date
    """
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)
    bundle_path = output_dir / "flow_import.zip"

    # Render each template
    workflow_json = engine.render_text("pa/workflow.json.j2", variables)
    api_props = engine.render_text("pa/apiProperties.json.j2", variables)
    definition = engine.render_text("pa/definition.json.j2", {
        **variables,
        "workflow_name": workflow_name,  # used in the action body
    })

    # Canonical PA paths
    definition_path_in_zip = f"Microsoft.Flow/flows/{workflow_name}/definition.json"

    with zipfile.ZipFile(bundle_path, "w", zipfile.ZIP_DEFLATED) as zf:
        zf.writestr("workflow.json", workflow_json)
        zf.writestr("apiProperties.json", api_props)
        zf.writestr(definition_path_in_zip, definition)

    return bundle_path
```

"""MCP package detection for auto-mode deploy.

Per D-02: Yigthinker NEVER calls MCP tools directly from workflow_deploy.
Auto mode only INSPECTS whether the expected MCP package is importable,
then returns instructional next_steps that tell the LLM caller which tool
to invoke on the MCP side.
"""
from __future__ import annotations

import importlib.util

# Target -> MCP package name mapping (per Phase 9 research Pattern 4)
MCP_PACKAGE_MAP: dict[str, str] = {
    "power_automate": "yigthinker_pa_mcp",
    "uipath": "yigthinker_uipath_mcp",
}

# Target -> suggested MCP tool + pip extras hint
MCP_TOOL_MAP: dict[str, dict[str, str]] = {
    "power_automate": {
        "suggested_tool": "power_automate_create_flow",
        "install_hint": "pip install yigthinker[pa-mcp]",
    },
    "uipath": {
        "suggested_tool": "uipath_publish_package",
        "install_hint": "pip install yigthinker[uipath-mcp]",
    },
}


def check_mcp_installed(package_name: str) -> bool:
    """Return True if the named MCP package is importable.

    Uses importlib.util.find_spec which does NOT import the module -
    cheap and side-effect-free.
    """
    try:
        return importlib.util.find_spec(package_name) is not None
    except (ModuleNotFoundError, ValueError):
        return False

```python
def cron_to_pa_recurrence(cron_expr: str) -> dict[str, int | str]:
    """Translate a cron expression to PA Recurrence {frequency, interval}.

    Per Pattern 7 research:
      "0 8 * * *"  -> frequency=Day,   interval=1
      "0 8 * * 1"  -> frequency=Week,  interval=1
      "0 8 5 * *"  -> frequency=Month, interval=1
      "0 */4 * * *"-> frequency=Hour,  interval=4

    For anything we can't map cleanly we fall back to Day/1 and set
    needs_manual_review - the setup_guide tells the user to fix it in PA.
    """
    from croniter import croniter  # validated already by workflow_deploy caller
    if not croniter.is_valid(cron_expr):
        raise ValueError(f"Invalid cron for PA mapping: {cron_expr}")
    parts = cron_expr.split()
    if len(parts) != 5:
        return {"frequency": "Day", "interval": 1, "needs_manual_review": True}
    minute, hour, dom, month, dow = parts

    # Hour: "0 */N * * *"
    if dom == "*" and month == "*" and dow == "*" and hour.startswith("*/"):
        try:
            return {"frequency": "Hour", "interval": int(hour[2:]), "needs_manual_review": False}
        except ValueError:
            pass

    # Day: "M H * * *"
    if dom == "*" and month == "*" and dow == "*" and "/" not in hour:
        return {"frequency": "Day", "interval": 1, "needs_manual_review": False}

    # Week: "M H * * D"
    if dom == "*" and month == "*" and dow != "*" and "/" not in dow and "-" not in dow:
        return {"frequency": "Week", "interval": 1, "needs_manual_review": False}

    # Month: "M H D * *"
    if dom != "*" and month == "*" and dow == "*" and "/" not in dom and "-" not in dom:
        return {"frequency": "Month", "interval": 1, "needs_manual_review": False}

    # Anything else - fall back
    return {"frequency": "Day", "interval": 1, "needs_manual_review": True}
```

    # Registry writeback

                "deploy_mode": "guided",

            v["status"] = "active"

    self._registry.save_manifest(input.workflow_name, updated)

    return ToolResult(
        tool_use_id="",
        content={
            "mode": "guided",
            "target": input.target,
            "workflow_name": input.workflow_name,
            "version": version,
            "deploy_id": deploy_id,
            "artifacts_ready": {
                "bundle": str(bundle_path),
                "setup_guide": str(setup_path),
            },
            "needs_manual_review": needs_manual_review,
            "message": (
                f"Guided bundle ready at {bundle_path}. "
                f"Open {setup_path} for step-by-step import instructions."
            ),
        },
    )
